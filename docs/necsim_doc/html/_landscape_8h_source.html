61  // The map files which are read in (or generated if running with "null" as the map file".
62  // Pristine maps are meant for before any deforestation occured, whereas the other maps are intended for modern day maps.
63  // A linear transformation from modern to pristine maps is used, approaching the habitat_change_rate variable times the difference between the pristine and modern maps.
64  // Once the gen_since_pristine number of generations has been reached, the map will jump to the pristine condition.
65  // the finer grid for the area around the sample area.
66  Map<uint32_t> fine_map;
67  // the pristine finer map.
68  Map<uint32_t> pristine_fine_map;
69  // the coarser grid for the wider zone.

72  Map<uint32_t> pristine_coarse_map;

77  // the maximum values for each dimension for offsetting.
78  long fine_x_max{}, fine_y_max{}, coarse_x_max{}, coarse_y_max{};
79  // the offsetting of the map in FINE map units.
80  long fine_x_offset{}, fine_y_offset{}, coarse_x_offset{}, coarse_y_offset{};
