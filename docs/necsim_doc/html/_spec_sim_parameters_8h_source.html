99  void setup(string file_in, bool use_spatial_in, string sample_file, const vector<double> &times,
100  string use_fragments_in, vector<double> speciation_rates, vector<double> min_speciation_gen_in,
101  vector<double> max_speciation_gen_in,
102  unsigned long metacommunity_size_in, double metacommunity_speciation_rate_in)
103  {
104  filename = std::move(file_in);
105  use_spatial = use_spatial_in;
106  samplemask = std::move(sample_file);
107  if(times.empty() && all_times.empty())
108  {
109  times_file = "null";
110  all_times.push_back(0.0);
111  }
112  else
113  {
114  times_file = "set";
115  for(const auto item : times)
116  {
117  all_times.emplace_back(item);
118  }
119  }
120  if(min_speciation_gen_in.size() != max_speciation_gen_in.size())
121  {
122  throw FatalException("Protracted parameter mismatch between minimum and maximum generations "
123  "(must be equal length).");
124  }
125  if(!protracted_parameters.empty())
126  {
127  protracted_parameters.clear();
128  }
129  for(unsigned long i = 0; i < min_speciation_gen_in.size(); i++)
130  {
131  ProtractedSpeciationParameters tmp{};
132  tmp.min_speciation_gen = min_speciation_gen_in[i];
133  tmp.max_speciation_gen = max_speciation_gen_in[i];
134  protracted_parameters.emplace_back(tmp);
135  }
136  use_fragments = !(use_fragments_in == "F");
137  fragment_config_file = use_fragments_in;
138  bMultiRun = speciation_rates.size() > 1;
139  for(auto speciation_rate : speciation_rates)
140  {
141  all_speciation_rates.push_back(speciation_rate);
142  }
143  metacommunity_size = metacommunity_size_in;
144  metacommunity_speciation_rate = metacommunity_speciation_rate_in;
145  }
