69  void setup(string file_in, bool use_spatial_in, string sample_file, const vector<double> &times,
70  const string &use_fragments_in, vector<double> speciation_rates)
71  {
72  filename = std::move(file_in);
73  use_spatial = use_spatial_in;
74  samplemask = std::move(sample_file);
75  if(times.empty() && all_times.empty())
76  {
77  times_file = "null";
78  all_times.insert(0.0);
79  }
80  else
81  {
82  times_file = "set";
83  for(const auto item : times)
84  {
85  all_times.insert(item);
86  }
87  }
88  use_fragments = !(use_fragments_in == "F");
89  fragment_config_file = use_fragments_in;
90  bMultiRun = speciation_rates.size() > 1;
91  for(auto speciation_rate : speciation_rates)
92  {
93  all_speciation_rates.insert(speciation_rate);
94  }
95  }
