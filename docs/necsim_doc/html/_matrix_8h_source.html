636  }

642  nBlockYSize = poDataset->GetRasterYSize();
643  double noDataValue;

646  noDataValue = poBand->GetNoDataValue();

663  }
664  // Check sizes match

670  {

679  ss << "Tif data type does not match object data size in " << filename << endl;
680  writeWarning(ss.str());
