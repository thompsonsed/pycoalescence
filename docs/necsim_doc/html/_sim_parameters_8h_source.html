428  bool setHistorical(unsigned int n)
429  {
430  is_historical = true;
431  bool finemapcheck = false;
432  bool coarsemapcheck = false;
433  // Loop over each element in the config file (each line) and check if it is historical fine or historical coarse.
434  for(unsigned int i = 0; i < configs.getSectionOptionsSize(); i ++ )
435  {
436  if(configs[i].section.find("historical_fine") == 0)
437  {
438  // Then loop over each element to find the number, and check if it is equal to our input number.
439  if(stol(configs[i].getOption("number")) == n)
440  {
441  is_historical = false;
442  string tmpmapfile;
443  tmpmapfile = configs[i].getOption("path");
444  if(historical_fine_map_file != tmpmapfile)
445  {
446  finemapcheck = true;
447  historical_fine_map_file = tmpmapfile;
448  }
449  habitat_change_rate = stod(configs[i].getOption("rate"));
450  gen_since_historical = stod(configs[i].getOption("time"));
451  }
452  }
453  else if(configs[i].section.find("historical_coarse") == 0)
454  {
455  if(stol(configs[i].getOption("number")) == n)
456  {
457  string tmpmapfile;
458  tmpmapfile = configs[i].getOption("path");
459  is_historical = false;
460  if(tmpmapfile != historical_coarse_map_file)
461  {
462  coarsemapcheck=true;
463  historical_coarse_map_file = tmpmapfile;
464  // check matches
465  if(habitat_change_rate != stod(configs[i].getOption("rate")) ||
466  gen_since_historical != stod(configs[i].getOption("time")))
467  {
468  writeWarning("Forest transform values do not match between fine and coarse maps. Using fine values.");
469  }
470  }
471  }
472  }
473  }
474  // if one of the maps has changed, we need to update, so return true.
475  if(finemapcheck != coarsemapcheck)
476  {
477  return true;
478  }
479  else
480  {
481  // finemapcheck should therefore be the same as coarsemapcheck
482  return finemapcheck;
483  }
