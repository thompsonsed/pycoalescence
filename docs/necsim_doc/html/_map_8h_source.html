248  bool importTif(const string &filename)
249  {
250 
251  if(filename.find(".tif") != string::npos)
252  {
253  stringstream ss;
254  ss << "Importing " << filename << " " << flush;
255  writeInfo(ss.str());
256  open(filename);
257  getRasterBand();
258  getBlockSizes();
259  getMetaData();
260  // If the sizes are 0 then use the raster sizes
261  if(numCols == 0 || numRows == 0)
262  {
263  setSize(blockYSize, blockXSize);
264  }
265  // Check sizes
266  if((numCols != blockXSize || numRows != blockYSize) || numCols == 0 ||
267  numRows == 0)
268  {
269  stringstream stringstream1;
270  stringstream1 << "Raster data size does not match inputted dimensions for " << filename
271  << ". Using raster sizes."
272  << endl;
273  stringstream1 << "Old dimensions: " << numCols << ", " << numRows << endl;
274  stringstream1 << "New dimensions: " << blockXSize << ", " << blockYSize << endl;
275  writeWarning(stringstream1.str());
276  setSize(blockYSize, blockXSize);
277  }
278  // Check the data types are support
279  const char *dt_name = GDALGetDataTypeName(dt);
280  if(dt == 0 || dt > 7)
281  {
282  throw FatalException("Data type of " + string(dt_name) + " is not supported.");
283  }
284 #ifdef DEBUG
285  if(sizeof(T) * 8 != gdal_data_sizes[dt])
286  {
287  stringstream ss2;
288  ss2 << "Object data size: " << sizeof(T) * 8 << endl;
289  ss2 << "Tif data type: " << dt_name << ": " << gdal_data_sizes[dt] << " bytes" << endl;
290  ss2 << "Tif data type does not match object data size in " << filename << endl;
291  writeWarning(ss2.str());
292  }
293 #endif
294  // Just use the overloaded method for importing between types
295  internalImport();
296  writeInfo("done!\n");
297  return true;
298  }
299  return false;
300  }
