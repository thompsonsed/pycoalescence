44 {
45  char *database;
46  int record_spatial;
47  char *sample_file;
48  char *fragment_file;
49  unsigned long metacommunity_size = 0;
50  double metacommunity_speciation_rate = 0.0;
51  PyObject *list_speciation_rates;
52  PyObject *list_times;
53  PyObject *list_min_spec_gen = nullptr;
54  PyObject *list_max_spec_gen = nullptr;
55  if(!PyArg_ParseTuple(args, "sissO!O!|O!O!kd", &database, &record_spatial, &sample_file,
56  &fragment_file, &PyList_Type, &list_speciation_rates, &PyList_Type, &list_times, &PyList_Type,
57  &list_min_spec_gen, &PyList_Type,
58  &list_max_spec_gen, &metacommunity_size, &metacommunity_speciation_rate))
59  {
60  return nullptr;
61  }
62  getGlobalLogger(self->logger, self->log_function);
63 #ifdef DEBUG
64  if(metacommunity_size == 0)
65  {
66  writeLog(10, "Metacommunity size not set.");
67  }
68  else
69  {
70  writeLog(10, "Metacommunity size: " + to_string(metacommunity_size));
71  }
72 #endif // DEBUG
73 
74  // Convert all our variables to the relevant form
75  string database_str = database;
76  auto use_spatial = static_cast<bool>(record_spatial);
77  string sample_file_str = sample_file;
78  string fragment_file_str = fragment_file;
79  vector<double> speciation_rates;
80  vector<double> times;
81  vector<double> min_speciation_gen;
82  vector<double> max_speciation_gen;
83  if(!importPyListToVectorDouble(list_speciation_rates, speciation_rates, "Speciation rates must be floats."))
84  {
85  return nullptr;
86  }
87  if(!importPyListToVectorDouble(list_times, times, "Times must be floats."))
88  {
89  return nullptr;
90  }
91  if(!list_min_spec_gen || !list_max_spec_gen)
92  {
93  min_speciation_gen.push_back(0.0);
94  max_speciation_gen.push_back(0.0);
95  }
96  else
97  {
98  if(!importPyListToVectorDouble(list_min_spec_gen, min_speciation_gen,
99  "Minimum protracted generations must be floats."))
100  {
101  return nullptr;
102  }
103  if(!importPyListToVectorDouble(list_max_spec_gen, max_speciation_gen,
104  "Maximum protracted generations must be floats."))
105  {
106  return nullptr;
107  }
108  }
109  try
110  {
111  self->specSimParameters->setup(std::move(database_str), use_spatial, sample_file_str, times,
112  fragment_file_str, speciation_rates, min_speciation_gen,
113  max_speciation_gen, metacommunity_size, metacommunity_speciation_rate);
114  }
115  catch(exception &e)
116  {
117  removeGlobalLogger();
118  PyErr_SetString(NECSimError, e.what());
119  return nullptr;
120  }
121  Py_RETURN_NONE;
122 }
