91  void setSeed(long seed)
92  {
93  if(!seeded)
94  {
95  idum2 = 123456789;
96  iy = 0;
97  idum = seed;
98  if(idum < 1) idum = 1;
99  //Be sure to prevent idum = 0.
100  idum2 = (idum);
101  for(j = NTAB + 7; j >= 0; j--)
102  {
103  //Load the shuffle table (after 8 warm-ups).
104  k = (idum) / IQ1;
105  idum = IA1 * (idum - k * IQ1) - k * IR1;
106  if(idum < 0) idum += IM1;
107  if(j < NTAB) iv[j] = idum;
108  }
109  iy = iv[0];
110  seeded = true;
111  }

114  throw runtime_error("Trying to set the seed again: this can only be set once.");

123  double d01()
124  {
125  k = (idum) / IQ1;
126  //Start here when not initializing.
127  idum = IA1 * (idum - k * IQ1) - k * IR1;
128  //Compute idum=(IA1*idum) % IM1 without overflows by Schrage's method.
129  if(idum < 0) idum += IM1;
130  k = idum2 / IQ2;
131  idum2 = IA2 * (idum2 - k * IQ2) - k * IR2;
132  //Compute idum2=(IA2*idum) % IM2 likewise.
133  if(idum2 < 0) idum2 += IM2;
134  j = iy / NDIV;
135  //Will be in the range 0..NTAB-1.
136  iy = iv[j] - idum2;
137  //Here idum is shuffled, idum and idum2 are combined to generate output.
138  iv[j] = idum;
139  if(iy < 1) iy += IMM1;
140  if((temp = AM * iy) > RNMX)
141  {
142  //os << "random call = " << "RNMAX" << "\n";
143  return RNMX; //Because users don't expect endpoint values.
144  }
145  return temp;

164  double norm()
165  {
166  if(normflag)
167  {
168  double r2 = 2;
169  double xx = 0;
170  double yy = 0;
171  while(r2 > 1)
172  {
173  xx = 2.0 * d01() - 1.0;
174  yy = 2.0 * d01() - 1.0;
175  r2 = (xx * xx) + (yy * yy);
176  }
177  double fac = sqrt(-2.0 * log(r2) / r2);
178  lastresult = xx * fac;
179  double result = yy * fac;
180  normflag = false;
181  return sigma * result;
182  }
183  else
184  {
185  normflag = true;
186  return sigma * lastresult;
187  }
188  }

334  void setDispersalMethod(const string &dispersal_method, const double &m_probin, const double &cutoffin)
335  {
336  if(dispersal_method == "normal")
337  {
338  dispersalFunction = &NRrand::norm2D;
339  if(sigma < 0)
340  {
341  throw invalid_argument("Cannot have negative sigma with normal dispersal");
342  }
343  }
344  else if(dispersal_method == "fat-tail" || dispersal_method == "fat-tailed")
345  {
346  dispersalFunction = &NRrand::fattail;
347  if(tau < 0 || sigma < 0)
348  {
349  throw invalid_argument("Cannot have negative sigma or tau with fat-tailed dispersal");
350  }
351  }
352  else if(dispersal_method == "norm-uniform")
353  {
354  dispersalFunction = &NRrand::normUniform;
355  if(sigma < 0)
356  {
357  throw invalid_argument("Cannot have negative sigma with normal dispersal");
358  }
359  }
360  else if(dispersal_method == "uniform-uniform")
361  {
362  // This is just here for testing purposes
363  dispersalFunction = &NRrand::uniformUniform;
364  }
365  // Also provided the old version of the fat-tailed dispersal kernel
366  else if(dispersal_method == "fat-tail-old")
367  {
368  dispersalFunction = &NRrand::fattail_old;
369  if(tau > -2 || sigma < 0)
370  {
371  throw invalid_argument(
372  "Cannot have sigma < 0 or tau > -2 with fat-tailed dispersal (old implementation).");
373  }
374  }
375  else
376  {
377  throw runtime_error("Dispersal method not detected. Check implementation exists");
378  }
379  m_prob = m_probin;
380  cutoff = cutoffin;
381  }
