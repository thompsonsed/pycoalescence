104  void setSeed(long seed)
105  {
106  if(!seeded)
107  {
108  idum2 = 123456789;
109  iy = 0;
110  idum = abs(seed);
111  if(idum < 1) idum = 1;
112  //Be sure to prevent idum = 0.
113  idum2 = idum;
114  for(j = NTAB + 7; j >= 0; j--)
115  {
116  //Load the shuffle table (after 8 warm-ups).
117  k = (idum) / IQ1;
118  idum = IA1 * (idum - k * IQ1) - k * IR1;
119  if(idum < 0) idum += IM1;
120  if(j < NTAB) iv[j] = idum;
121  }
122  iy = iv[0];
123  seeded = true;
124  }
125  else
126  {
127  throw runtime_error("Trying to set the seed again: this can only be set once.");
128  }
129  }

145  double d01()
146  {
147  k = (idum) / IQ1;
148  //Start here when not initializing.
149  idum = IA1 * (idum - k * IQ1) - k * IR1;
150  //Compute idum=(IA1*idum) % IM1 without overflows by Schrage's method.
151  if(idum < 0) idum += IM1;
152  k = idum2 / IQ2;
153  idum2 = IA2 * (idum2 - k * IQ2) - k * IR2;
154  //Compute idum2=(IA2*idum) % IM2 likewise.
155  if(idum2 < 0) idum2 += IM2;
156  j = iy / NDIV;
157  //Will be in the range 0..NTAB-1.
158  iy = iv[j] - idum2;
159  //Here idum is shuffled, idum and idum2 are combined to generate output.
160  iv[j] = idum;
161  if(iy < 1) iy += IMM1;
162  if((temp = AM * iy) > RNMX)
163  {
164  return RNMX; //Because users don't expect endpoint values.
165  }
166  return temp;
167 
168  }

185  double norm()
186  {
187  if(normflag)
188  {
189  double r2 = 2;
190  double xx = 0;
191  double yy = 0;
192  while(r2 > 1)
193  {
194  xx = 2.0 * d01() - 1.0;
195  yy = 2.0 * d01() - 1.0;
196  r2 = (xx * xx) + (yy * yy);
197  }
198  double fac = sqrt(-2.0 * log(r2) / r2);
199  lastresult = xx * fac;
200  double result = yy * fac;
201  normflag = false;
202  return sigma * result;
203  }
204  else
205  {
206  normflag = true;
207  return sigma * lastresult;
208  }
209  }

226  double rayleighMinDist(const double &dist)
227  {
228  double min_prob = rayleighCDF(dist);
229  double rand_prob = min_prob + (1 - min_prob) * d01();
230  double out = sigma * pow(-2 * log(rand_prob), 0.5);
231  if(out < dist)
232  {
233  // This probably means that the rayleigh distribution has a less-than-machine-precision probability of
234  // producing this distance.
235  // Therefore, we just return the distance
236  return dist;
237  }
238  return out;
239  }

465  void setDispersalMethod(const string &dispersal_method, const double &m_probin, const double &cutoffin)
466  {
467  if(dispersal_method == "normal")
468  {
469  dispersalFunction = &NRrand::rayleigh;
470  dispersalFunctionMinDistance = &NRrand::rayleighMinDist;
471  if(sigma < 0)
472  {
473  throw invalid_argument("Cannot have negative sigma with normal dispersal");
474  }
475  }
476  else if(dispersal_method == "fat-tail" || dispersal_method == "fat-tailed")
477  {
478  dispersalFunction = &NRrand::fattail;
479  dispersalFunctionMinDistance = &NRrand::fattailMinDistance;
480  if(tau < 0 || sigma < 0)
481  {
482  throw invalid_argument("Cannot have negative sigma or tau with fat-tailed dispersal");
483  }
484  }
485  else if(dispersal_method == "norm-uniform")
486  {
487  dispersalFunction = &NRrand::normUniform;
488  dispersalFunctionMinDistance = &NRrand::normUniformMinDistance;
489  if(sigma < 0)
490  {
491  throw invalid_argument("Cannot have negative sigma with normal dispersal");
492  }
493  }
494  else if(dispersal_method == "uniform-uniform")
495  {
496  // This is just here for testing purposes
497  dispersalFunction = &NRrand::uniformUniform;
498  dispersalFunctionMinDistance = &NRrand::uniformUniformMinDistance;
499  }
500  // Also provided the old version of the fat-tailed dispersal kernel
501  else if(dispersal_method == "fat-tail-old")
502  {
503  dispersalFunction = &NRrand::fattail_old;
504  dispersalFunctionMinDistance = &NRrand::fattailMinDistance;
505 
506  if(tau > -2 || sigma < 0)
507  {
508  throw invalid_argument(
509  "Cannot have sigma < 0 or tau > -2 with fat-tailed dispersal (old implementation).");
510  }
511  }
512  else
513  {
514  throw runtime_error("Dispersal method not detected. Check implementation exists");
515  }
516  m_prob = m_probin;
517  cutoff = cutoffin;
518  }

553  unsigned long randomLogarithmic(long double alpha)
554  {
555  double u_2 = d01();
556  if(u_2 > alpha)
557  {
558  return 1;
559  }
560  long double h = log(1 - alpha);
561  long double q = 1 - exp(d01() * h);
562  if(u_2 < (q * q))
563  {
564  return static_cast<unsigned long>(floor(1 + log(u_2) / log(q)));
565  }
566  else if(u_2 > q)
567  {
568  return 1;
569  }
570  else
571  {
572  return 2;
