109  void setSeed(long seed)
110  {
111  if(!seeded)
112  {
113  idum2 = 123456789;
114  iy = 0;
115  idum = seed;
116  if(idum < 1) idum = 1;
117  //Be sure to prevent idum = 0.
118  idum2 = (idum);
119  for(j = NTAB + 7; j >= 0; j--)
120  {
121  //Load the shuffle table (after 8 warm-ups).
122  k = (idum) / IQ1;
123  idum = IA1 * (idum - k * IQ1) - k * IR1;
124  if(idum < 0) idum += IM1;
125  if(j < NTAB) iv[j] = idum;
126  }
127  iy = iv[0];
128  seeded = true;
129  }
130  else
131  {
132  throw runtime_error("Trying to set the seed again: this can only be set once.");
133  }
134  }

150  double d01()
151  {
152  k = (idum) / IQ1;
153  //Start here when not initializing.
154  idum = IA1 * (idum - k * IQ1) - k * IR1;
155  //Compute idum=(IA1*idum) % IM1 without overflows by Schrage's method.
156  if(idum < 0) idum += IM1;
157  k = idum2 / IQ2;
158  idum2 = IA2 * (idum2 - k * IQ2) - k * IR2;
159  //Compute idum2=(IA2*idum) % IM2 likewise.
160  if(idum2 < 0) idum2 += IM2;
161  j = iy / NDIV;
162  //Will be in the range 0..NTAB-1.
163  iy = iv[j] - idum2;
164  //Here idum is shuffled, idum and idum2 are combined to generate output.
165  iv[j] = idum;
166  if(iy < 1) iy += IMM1;
167  if((temp = AM * iy) > RNMX)
168  {
169  //os << "random call = " << "RNMAX" << "\n";
170  return RNMX; //Because users don't expect endpoint values.
171  }
172  return temp;
173 
174  }

191  double norm()
192  {
193  if(normflag)
194  {
195  double r2 = 2;
196  double xx = 0;
197  double yy = 0;
198  while(r2 > 1)
199  {
200  xx = 2.0 * d01() - 1.0;
201  yy = 2.0 * d01() - 1.0;
202  r2 = (xx * xx) + (yy * yy);
203  }
204  double fac = sqrt(-2.0 * log(r2) / r2);
205  lastresult = xx * fac;
206  double result = yy * fac;
207  normflag = false;
208  return sigma * result;
209  }
210  else
211  {
212  normflag = true;
213  return sigma * lastresult;
214  }
215  }

361  void setDispersalMethod(const string &dispersal_method, const double &m_probin, const double &cutoffin)
362  {
363  if(dispersal_method == "normal")
364  {
365  dispersalFunction = &NRrand::norm2D;
366  if(sigma < 0)
367  {
368  throw invalid_argument("Cannot have negative sigma with normal dispersal");
369  }
370  }
371  else if(dispersal_method == "fat-tail" || dispersal_method == "fat-tailed")
372  {
373  dispersalFunction = &NRrand::fattail;
374  if(tau < 0 || sigma < 0)
375  {
376  throw invalid_argument("Cannot have negative sigma or tau with fat-tailed dispersal");
377  }
378  }
379  else if(dispersal_method == "norm-uniform")
380  {
381  dispersalFunction = &NRrand::normUniform;
382  if(sigma < 0)
383  {
384  throw invalid_argument("Cannot have negative sigma with normal dispersal");
385  }
386  }
387  else if(dispersal_method == "uniform-uniform")
388  {
389  // This is just here for testing purposes
390  dispersalFunction = &NRrand::uniformUniform;
391  }
392  // Also provided the old version of the fat-tailed dispersal kernel
393  else if(dispersal_method == "fat-tail-old")
394  {
395  dispersalFunction = &NRrand::fattail_old;
396  if(tau > -2 || sigma < 0)
397  {
398  throw invalid_argument(
399  "Cannot have sigma < 0 or tau > -2 with fat-tailed dispersal (old implementation).");
400  }
401  }
402  else
403  {
404  throw runtime_error("Dispersal method not detected. Check implementation exists");
405  }
406  m_prob = m_probin;
407  cutoff = cutoffin;
408  }
