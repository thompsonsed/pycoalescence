   void DataPoint::decreaseNwrap()
   {
       if(nwrap == 0)
       {
           throw out_of_range("ERROR_DATA_001: Trying to decrease  nwrap less than 0.");
       }
       else
       {
           nwrap--;
       }
   }

   }
